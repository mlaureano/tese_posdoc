%%%% LISTA DE SÍMBOLOS (ELEMENTO OPCIONAL; ENTRADAS)
%%
%% Conjunto de sinais que substituem o nome de uma coisa ou de uma ação.
%% Elaborada conforme a ordem apresentada no texto, com o devido significado.
%%
%% Notas:
%% 1. Sem subdivisões (\MakeSymbols em ./utfpr-thesis.tex), os itens são
%%    ordenados pela ocorrência (impressão no texto) em uma única lista.
%% 2. Com subdivisões (\MakeSymbols* em ./utfpr-thesis.tex), os itens são
%%    agrupados por tipo e ordenados pelo rótulo (label).
%%
%% Definição de uma entrada:
%% \New<SymbolType>Entry{Label}{%% Usar somente letras latinas não acentuadas
%%   Term        = {...},%       % Obrigatório
%%   Description = {...},%       % Obrigatório
%%   Unit        = {...},%       % Opcional
%%   Sort        = {...},%       % Opcional (para reordenar na lista)
%% }
%% Se a letra inicial da descrição for acentuada, deve ser colocada entre
%% chaves, por exemplo, Description = {{á}rea}.
%%
%% Impressão no texto e adição (automática) na lista (somente termo):
%% +========================+==========================================+
%% | Comando                | Imprime                                  |
%% +------------------------+------------------------------------------+
%% | \<symtype>{Label}      | Termo                                    |
%% | \<symtype>descr{Label} | Descrição                                |
%% | \<SymType>Descr{Label} | Descrição com letra inicial em maiúscula |
%% | \<symtype>unit{Label}  | Unidade                                  |
%% +========================+==========================================+
%%
%% Adição (automática) também no Índice Remissivo com um asterisco (*) opcional:
%% +====================+=========+
%% | Comando            | Imprime |
%% +--------------------+---------+
%% | \<symtype>*{Label} | Termo   |
%% +====================+=========+
%%
%% <SymbolType>, <symtype> e <SymType> são definidos conforme:
%% +==============+==============+===========+===========+
%% | Tipo         | <SymbolType> | <symtype> | <SymType> |
%% +--------------+--------------+-----------+-----------+
%% | Notação (¹)  | Notation     | nttn      | Nttn      |
%% | Sobrescrito  | Superscript  | sprs      | Sprs      |
%% | Subscrito    | Subscript    | sbsc      | Sbsc      |
%% | Letra grega  | GreekLetter  | grkl      | GrkL      |
%% | Letra latina | LatinLetter  | ltnl      | LtnL      |
%% +==============+==============+===========+===========+
%% (¹) o comando (termo) armazena um símbolo no comando \MrkSym (\DottedCircle
%% por padrão, no qual se aplica a notação durante a impressão da mesma) com um
%% segundo argumento (opcional: \nttn{Label}[MrkSym]).
